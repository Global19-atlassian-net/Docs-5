<sample Light>
cfg.Light
cfg.MUI

function OnStart()
{
    color = MUI.colors.teal
    app.InitializeUIKit(color.teal, "Light")

    lay = MUI.CreateLayout("Linear", "VCenter,FillXY")
    	
    	lst = "Apple,Mango,Orange,Banana"
    	chk = MUI.CreateCheckbox(lst, 0.7)
    	lay.AddChild(chk)

    app.AddLayout(lay)
}
</sample>

<sample Dark>
cfg.Light
cfg.MUI

function OnStart()
{
    color = MUI.colors.teal
    app.InitializeUIKit(color.teal, "Dark")

    lay = MUI.CreateLayout("Linear", "VCenter,FillXY")
    	
    	lst = "Apple,Mango,Orange,Banana"
    	chk = MUI.CreateCheckbox(lst, 0.7)
    	chk.SetOnTouch(OnCheckBox)
    	lay.AddChild(chk)

    app.AddLayout(lay)
}

function OnCheckBox(index, value)
{
	app.ShowPopup(lst[index] + " is "+value)
}
</sample>
