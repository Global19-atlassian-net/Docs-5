<sample Change Layout Background>
cfg.Light
cfg.MUI

function OnStart()
{

    app.InitializeUIKit("#009688", "Light")
    color = MUI.colors.teal

    lay = MUI.CreateLayout("linear", "FillXY,VCenter")

        btn = MUI.CreateButtonRaised("Choose Color", 0.4)
        btn.SetOnTouch(ShowDialog)
        lay.AddChild(btn)

    app.AddLayout(lay)

    var choices = [
        {name: "lighten4"},
        {name: "lighten3"},
        {name: "lighten2"},
        {name: "lighten1"},
        {name: "teal"},
        {name: "darken1"},
        {name: "darken2"},
        {name: "darken3"},
        {name: "darken4"}
    ]
    lsd = MUI.CreateListDialog("Select Color", choices)
    lsd.SetOnSelect(OnSelect)
}

function ShowDialog()
{
    lsd.Show()
}

function OnSelect(name)
{
    lay.SetBackColor(color[name])
}
</sample>
