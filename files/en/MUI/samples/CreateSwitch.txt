<sample Basic>
cfg.Light
cfg.MUI

function OnStart()
{
    color = MUI.colors.teal
    app.InitializeUIKit(color.teal, "Light")

    lay = MUI.CreateLayout("Linear", "FillXY,VCenter")

        skb = MUI.CreateSwitch()
        skb.SetOnTouch(OnTouch)
        lay.AddChild(skb)

    app.AddLayout(lay)
}

function OnTouch(value) 
{
    app.ShowPopup(value)
}
</sample>
